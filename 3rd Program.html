Java Program: Custom Exception Handling

        
public class PayoutOutOfBoundsException extends Exception {
    // Constructor to accept an error message
    public PayoutOutOfBoundsException(String message) {
        super(message);
    }
}

// Class to demonstrate throwing the custom exception
public class PayoutDemo {
    // Method to validate payout and throw exception if out of bounds
    public static void validatePayout(double payout) throws PayoutOutOfBoundsException {
        if (payout < 0 || payout > 10000) {
            throw new PayoutOutOfBoundsException("Payout is out of bounds! Valid range is 0 to 10,000.");
        }
        System.out.println("Payout of Rs." + payout + " is valid.");
    }

    public static void main(String[] args) {
        double payout1 = 5000; // Valid payout
        double payout2 = 15000; // Invalid payout
        try {
            // Validate first payout
            validatePayout(payout1);
            validatePayout(payout2);
        } catch (PayoutOutOfBoundsException e) {
            // Catch and handle the custom exception
            System.out.println("Exception caught: " + e.getMessage());
        }
        System.out.println("Program execution completed.");
    }
}
